\noindent
\textbf{Divergence Theorum and Curl}\\

\noindent
Gradient - Scalar Field \\
Divergence - Vector Field \\

\noindent
If \begin{math} \nabla \cdot \vec{A} = 0  \end{math} then \begin{math} \vec{A} = \end{math} solenoidal vector fied (whatever goes in comes out) \\

\noindent
Divergence Theorum

\begin{displaymath}\int\limits_V \nabla \vec{A} \cdot dV = \oint\limits_S \vec{A} \vec{dS} \end{displaymath}
\begin{displaymath} \int\limits_V \nabla \vec{D} \cdot dV = \oint\limits_S \vec{D} \vec{dS} \end{displaymath}

\noindent
Circulation of $\vec{A}$ around countour C

\begin{displaymath}
\Delta \oint\limits_C \vec{A} \vec{dl}
\end{displaymath}
*** $\Delta$ is underlined in Nguyen's notation. Look into this. \\

%%%%
\noindent
Curl
\begin{displaymath}
\nabla \times \vec{A} = \frac{1}{\Delta s} \lim_{\Delta s \to \O} \oint\limits_S \vec{A} \vec{dl} 
\end{displaymath}

\noindent
Stoke's Theorum

\begin{displaymath}
\int\limits_S \nabla \times \vec{A} \cdot \vec{ds} = \oint\limits_C \vec{A} \cdot \vec{dl}
\end{displaymath}

\noindent
Two Null Identitites

\begin{displaymath}
\nabla \times (\nabla V) \equiv 0
\end{displaymath}

\begin{displaymath}
\nabla \cdot (\nabla \vec{A}) \equiv 0
\end{displaymath}

\noindent
\textbf{Maxwell's Equations}

\begin{displaymath}
\vec{E}(x,y,z,t), \vec{H}, \vec{D}, \vec{B}, \vec{J}, \varphi_{v} 
\end{displaymath}

\noindent
Faraday
\begin{displaymath}
\nabla \times \vec{E} = \frac{- \delta \vec{B}}{\delta t}
\end{displaymath}

\noindent
Ampere
\begin{displaymath}
\nabla \times \vec{H} = \vec{J} + \frac{- \delta \vec{D}}{\delta t}
\end{displaymath}

\noindent
Gauss
\begin{displaymath}
\nabla \cdot \vec{D} = \varphi_{v}
\end{displaymath}

\noindent
No Magnetic Charges
\begin{displaymath}
\nabla \cdot \vec{B} = 0
\end{displaymath}

\noindent
Modified Ampere's Law
\begin{displaymath}
\frac{\delta \vec{D}}{\delta t}
\end{displaymath}

\noindent
\textbf{Special Cases of Maxwell's Equations: Static $\vec{E}$ and $\vec{H}$}

\begin{displaymath}
\vec{E}, \vec{H} = f(x,y,z) \neq f(t) \rightarrow \frac{\delta}{\delta t} = 0
\end{displaymath}

\noindent
Faraday
\begin{displaymath}
\nabla \times \vec{E} = 0 \qquad \qquad \qquad \oint\limits_C \vec{E} \cdot \vec{dl} = 0
\end{displaymath}

\noindent
Ampere
\begin{displaymath}
\nabla \times \vec{H} = \vec{J} \qquad \qquad \qquad \oint\limits_C \vec{H} \cdot \vec{dl} = I
\end{displaymath}

\noindent
Gauss
\begin{displaymath}
\nabla \times \vec{D} = \varphi_{v} \qquad \qquad \qquad \oint\limits_C \vec{D} \cdot \vec{ds} = Q
\end{displaymath}

\noindent
No Magnetic Charges
\begin{displaymath}
\nabla \times \vec{B} = 0 \qquad \qquad \qquad \oint\limits_C \vec{B} \cdot \vec{ds} = 0
\end{displaymath}

\noindent
Modified Ampere's Law
\begin{displaymath}
\frac{\delta \vec{D}}{\delta t}
\end{displaymath}
